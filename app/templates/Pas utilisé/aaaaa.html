'contain' : 'cover'; // Détecter les URLs problématiques (apostrophes, parenthèses) function hasProblematicChars(url) { return url.includes('%27') || url.includes('%28') || url.includes('%29') || url.includes('(') || url.includes(')') || url.includes("'"); } // MAX 3 tentatives avant d'abandonner if (errorCount >= 3) { console.log('Abandon après 3 tentatives:', img.dataset.original); img.style.display = 'none'; const placeholder = img.nextElementSibling; if (placeholder) placeholder.style.display = 'flex'; return; } // Incrémenter le compteur img.dataset.errorCount = (errorCount + 1).toString(); // STRATÉGIE DE FALLBACK à 3 niveaux const originalUrl = img.dataset.original; // Niveau 1: Si c'est une URL problématique, essayer wsrv.nl directement if (errorCount === 0 && hasProblematicChars(originalUrl)) { console.log('Tentative 1 (wsrv pour URL problématique):', originalUrl); img.src = `https://wsrv.nl/?url=${encodeURIComponent(originalUrl)}&w=250&h=250&fit=${fit}&output=webp&q=50`; img.dataset.currentService = 'wsrv'; return; } // Niveau 1 bis: Si c'est une URL normale, essayer weserv (comme avant) if (errorCount === 0) { console.log('Tentative 1 (weserv):', originalUrl); img.src = `https://images.weserv.nl/?url=${encodeURIComponent(originalUrl)}&w=250&h=250&fit=${fit}&output=webp&q=50`; img.dataset.currentService = 'weserv'; return; } // Niveau 2: Essayer wsrv.nl (plus tolérant) if (errorCount === 1) { console.log('Tentative 2 (wsrv):', originalUrl); img.src = `https://wsrv.nl/?url=${encodeURIComponent(originalUrl)}&w=250&h=250&fit=${fit}&output=webp&q=50`; img.dataset.currentService = 'wsrv'; return; } // Niveau 3: Dernier recours - URL directe sans service if (errorCount === 2) { console.log('Tentative 3 (URL directe):', originalUrl);
